-- Q01: For parameter PM2, show Exceed Days (any) by Country x Month for Q1 of 2024.
SELECT
  -- Get first 3 months (Jan, Feb, Mar) descending from 2024
  HEAD(
    DESCENDANTS(
      [dim_timemonth].[Time].[year_num].&[2024],
      [dim_timemonth].[Time].[month_name]
    ), 
    3
  ) ON COLUMNS,
  NON EMPTY 
    [dim_city].[Geo].[country_name].Members 
  ON ROWS
FROM [AirQ Cube]
WHERE (
  [dim_param].[Param].[param_name].&[PM2],
  [Measures].[Exceed Days (any)]
)
